#include <math.h>
using namespace std;

int main ()
{
    float sala, con1, con2, jc1, jc2,conj1,conj2;
    sala = sala + 1200.00;
    con1 = con1+ 200.00;
    con2 = con2+ 120.00;
    jc1 = (con1 * 2) / 100;
    jc2 = (con2 * 2) / 100;
    conj1 = jc1+con1;
    conj2 = jc2+con2;
    cout << "Joao, voce precisa pagar as seguintes contas:\n\n"<<
     "R$:"<< con1<< "\nR$:"<<con2<< "\n\nSomando os juros, respectivamente, totaliza:\n\n"<<

     <<"Este sera seu saldo apos os pagamentos:\n"<<"R$:"<<(sala-conj1)-conj2;
